************ CRIANDO TABELAS ***************

create table Empregado(
Enome varchar(80),
CPF int,
Endereço varchar(100),
Nasc date,
Sexo varchar(10),
Salário real,
Chefe int,
Cdep int,
primary key(CPF)
	)

create table Departamento(
Dnome varchar(80),
Codigo int,
Gerente int,
primary key (Codigo)	
	)

create table Projeto(
Pnome varchar(80),
PCodigo varchar(50),
Cidade varchar(80),
Cdep integer,	
primary key (PCodigo),
FOREIGN KEY (Cdep) REFERENCES Departamento (Codigo))


ALTER TABLE Empregado ADD FOREIGN KEY (Cdep) REFERENCES Departamento (Codigo)


create table DUnidade(
DCodigo int,
DCidade varchar(50),
primary key (DCidade),
FOREIGN KEY (DCodigo) REFERENCES Departamento (Codigo)	
)


create table Tarefa(
CPF int,
PCodigo varchar(50),
Horas real,
PRIMARY KEY (CPF, PCodigo),
FOREIGN KEY (CPF) REFERENCES Empregado (CPF),	
FOREIGN KEY (PCodigo) REFERENCES Projeto (PCodigo)
)

1. Recuperar para cada departamento: o seu nome, o maior e o menor salário recebido por empregados do departamento e a média salarial do departamento. 
select D.dnome, Max(E.salário), Min(E.salário), AVG(E.salário)
from empregado E, departamento D
where E.cdep=D.codigo
group by (D.dnome)

2. Recuperar o nome do departamento com maior média salarial. 
select d.dnome
	from empregado e, departamento d
	where d.dcodigo = e.cdep
	group by d.dnome
	having avg(e.salario) >= all(
					select avg(e.salario)
					from empregado e
					group by (cdep)
					)

3. Recuperar para cada departamento: o seu nome, o nome do seu gerente, a quantidade de empregados, a quantidade de projetos do departamento e a quantidade de unidades do departamento. 
select e1.enome as Gerente,
	       d.dnome as departamento,
	       count(distinct e2.CPF)    as num_funcionarios,
	       count(distinct p.pcodigo) as num_projeto,
	       count(distinct u)         as unidades

	from empregado e1,
	     departamento d,
	     empregado e2,
	     projeto p,
	     dunidade u
	
	where e1.CPF   = d.gerente
	  and e2.cdep  = d.codigo
	  and p.cdep   = d.codigo
	  and u.dcodigo = d.codigo

	group by (e1.enome, d.dnome)



4. Recuperar o nome do projeto que consome o maior número de horas. 
select p.pnome
from projeto p, tarefa t
where p.pcodigo = t.pcodigo
	group by (p.pnome)
	having sum(t.horas) >= all(
				select sum(t.horas)
				from tarefa t
				group by (t.pcodigo)
				)
5. Recuperar o nome do projeto mais caro. 
select p.pnome
	from tarefa t, 
	     projeto p,
	     (select e.CPF, e.salário/40 as sal_hora
	      from empregado e) as sal
						 
	where  p.pcodigo = t.pcodigo and t.CPF = sal.CPF
	group by (p.pnome)
	having (sum(t.horas*sal.sal_hora)) >= all(
						select sum(t.horas*sal.sal_hora)
						from tarefa t,
						     (select e.CPF, e.salário/40 as sal_hora
						      from empregado e) as sal
						where  t.CPF = sal.CPF
						group by (t.pcodigo)
						)

12. Listar nome dos departamentos com média salarial maior que a média salarial da empresa. 
select d.dnome
	from departamento d, empregado e
	where d.codigo = e.cdep
	group by (d.dnome)
	having avg(e.salário) > (
				select avg(e.salário)
				from empregado e
				)

13. Listar todos os empregados que possuem salário maior que a média salarial de seus departamentos. 
select e.enome, e.salário, ct.media_dep
	from empregado e, (
			select d.codigo, avg(e.salário) as media_dep
			from departamento d, empregado e
			where d.codigo = e.cdep
			group by (d.codigo)
			) as ct

	where ct.codigo = e.cdep and e.salário > ct.media_dep


14. Listar os empregados lotados nos departamentos localizados em “Fortaleza”. 
select e.enome, d.dnome, u.dcidade
	from empregado e, departamento d, dunidade u
	where e.cdep = d.codigo and d.codigo = u.dcodigo and u.dcidade = 'Fortaleza'

15. Listar nome de departamentos com empregados ganhando duas vezes mais que a média do departamento. 
select DISTINCT d.dnome 
from departamento d, empregado e,
     (select Cdep, avg(salário) as media
      from Empregado
				group by Cdep) as tab
where d.codigo = e.cdep and e.cdep = tab.cdep
             and e.salário > 2*tab.media
